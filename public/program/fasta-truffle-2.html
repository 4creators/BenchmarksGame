def make_repeat_fasta(id, desc, src, n)
    puts ">#{id} #{desc}"
    l = src.length
    s = src * ((n / l) + 1)
    s.slice!(n, l)
    0.step(s.length-1,60) {|x| print s[x,60] , "\n"}
end

def make_random_fasta(id, desc, table, n)
    puts ">#{id} #{desc}"
    rand, v = nil,nil
    prob = 0.0
    table.each{|v| v[1]= (prob += v[1])}
    output = ""
    n.times do
      rand = gen_random(1.0)
      table.each do |v|
	if v[1] > rand then
	  output << v[0]
	  break
	end
      end
    end
    0.step(output.length-1,60) {|x| print output[x,60] , "\n"}
end
